 安装新的软件包并删除冲突的软件包 sudo apt-get dist-upgrade

  sudo apt-get install curl
  sudo apt-get install git-core
  	 a、安装ssh，因为git是基于ssh协议的，所以必须先装ssh：
		sudo apt-get install openssh-server openssh-client
     b、安装好ssh后，启动ssh服务：
		sudo /etc/init.d/ssh restart     
     c、安装git服务器：
		sudo apt-get install git-core
	 d、git配置
	      git config --global user.name "Your Name"
        git config --global user.email your-email@address.com
        git config --global url."https://".insteadof git://
	 e、配置ssh公钥(首先在本地生成ssh公钥)
		    ssh-keygen -C 'your-email@address.com' -t rsa

1.安装RVM环境
  gpg --keyserver hkp://keys.gnupg.net --recv-keys 409B6B1796C275462A1703113804BB82D39DC0E3
  curl -L https://raw.githubusercontent.com/wayneeseguin/rvm/master/binscripts/rvm-installer | bash -s stable
2.载入RVM环境
  source ~/.rvm/scripts/rvm
  *将[[ -s "$HOME/.rvm/scripts/rvm" ]] && source "$HOME/.rvm/scripts/rvm" # Load RVM into a shell session *as a function*写到.bashrc文件中启动终端时自动载入
3.修改RVM下载Ruby的源，到Ruby China的镜像
  echo "ruby_url=https://cache.ruby-china.org/pub/ruby" > ~/.rvm/user/db
4.用RVM安装Ruby环境
  rvm requirements
  rvm install 2.3.0
5.设置Ruby版本
  rvm use 2.3.0 --default
6.修改gem源
  gem sources --add https://gems.ruby-china.org/ --remove https://rubygems.org/
7.安装Bundler
  gem install bundler
8.安装Rails环境
  gem install rails
9.rails server 启动webrick服务Could not find a JavaScript runtime报错处理
  gem install execjs
  gem install therubyracer
  sudo apt-get install nodejs


#安装MVM环境
 1.git clone https://github.com/creationix/nvm.git ~/.nvm && cd ~/.nvm && git checkout `git describe --abbrev=0 --tags`  
 2.cd后  vim .bashrc
 3.键入i，写入source ~/.nvm/nvm.sh ，键入esc->:wq保存退出
 4.将新增的nvm添加到系统中 source .bashrc
 5.查看可用的node版本 nvm ls-remote
 6.nvm install v6.4.0（自动安装npm，node package manage）
 7.nvm alias default 6.4.0
 8.npm install -g redis（安装Redis的Node.js客户端，这样Node.js程序就可以连接到Redis服务器并进行操作）
 	*需要先安装redis服务器 sudo apt-get install redis-server
 9.找不到node和npm报错处理
 	export NVM_DIR="/Users/YOURUSERNAME/.nvm"  
	[ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh"  # This loads nvm
	写入.bashrc并添加到系统中source .bashrc
